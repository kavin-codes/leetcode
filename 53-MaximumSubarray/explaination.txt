class Solution:
    def maxSubArray(self, nums):
        # Initialize with first element
        current_sum = nums[0]
        max_sum = nums[0]

        # Traverse the list
        for num in nums[1:]:
            # Choose maximum between starting new or continuing
            current_sum = max(num, current_sum + num)

            # Update maximum sum
            max_sum = max(max_sum, current_sum)

        return max_sum


# 🔽 Function call
nums = [-2, 1, -3, 4, -1, 2, 1, -5, 4]

obj = Solution()              # create object
result = obj.maxSubArray(nums)  # call function

print("Maximum Subarray Sum:", result)

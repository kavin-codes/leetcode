class Solution:
    def intersection(self, nums1: List[int], nums2: List[int]) -> List[int]:
        # Step 1: Create dictionaries to store frequency of elements in both arrays
        freq1 = {}  # Dictionary to store count of each number in nums1
        freq2 = {}  # Dictionary to store count of each number in nums2

        # Step 2: Count frequency of numbers in nums1
        for num in nums1:
            if num in freq1:        # If number already exists in dictionary, increment its count
                freq1[num] += 1
            else:                   # If number does not exist, initialize count to 1
                freq1[num] = 1

        # Step 3: Count frequency of numbers in nums2
        for num in nums2:
            if num in freq2:        # If number already exists in dictionary, increment its count
                freq2[num] += 1
            else:                   # If number does not exist, initialize count to 1
                freq2[num] = 1

        # Step 4: Find intersection elements
        res = []                    # Initialize an empty list to store the result
        for answer in freq1:        # Loop through all unique numbers in nums1
            if answer in freq2:     # If the number also exists in nums2, it's part of intersection
                res.append(answer)  # Add it to the result list

        # Step 5: Return the final intersection list
        return res

s = "loveleetcode"   # Input string

class Solution:
    def firstUniqChar(self, s: str) -> int:
        d = {}        # Dictionary to store frequency of each character

        # Loop through each character in the string
        # and count how many times it appears
        for ch in s:
            if ch in d:
                d[ch] += 1   # If character already exists, increase count
            else:
                d[ch] = 1    # If character is new, set count to 1

        # Loop through the string again with index
        # to find the first character with count = 1
        for i, ch in enumerate(s):
            if d[ch] == 1:   # Check if character appears only once
                return i    # Return its index

        return -1            # If no unique character found, return -1
    

s1 = Solution()                 # Create object of Solution class
print(s1.firstUniqChar(s))      # Call function and print result

---------------------------------------------------------------------------------------------

✅ Two-Pass Hash Map (Frequency Count) Pattern

The solution uses:

First pass → count how many times each character appears

Second pass → find the first character whose count is 1

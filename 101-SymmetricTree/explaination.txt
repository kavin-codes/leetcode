# Definition for a binary tree node
class TreeNode:
    def __init__(self, val=0, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right

class Solution:
    def isSymmetric(self, root):

        # Function to check if two trees are mirror images
        def mirror(t1, t2):

            # If both nodes are None → symmetric
            if not t1 and not t2:
                return True

            # If only one node exists → not symmetric
            if not t1 or not t2:
                return False

            # Check values and recursively compare mirrored children
            return (t1.val == t2.val and
                    mirror(t1.left, t2.right) and
                    mirror(t1.right, t2.left))

        # Compare the tree with itself
        return mirror(root, root)


# Create sample tree
root = TreeNode(1)
root.left = TreeNode(2)
root.right = TreeNode(2)
root.left.left = TreeNode(3)
root.left.right = TreeNode(4)
root.right.left = TreeNode(4)
root.right.right = TreeNode(3)

# Function call
solution = Solution()
result = solution.isSymmetric(root)

# Print result
print(result)

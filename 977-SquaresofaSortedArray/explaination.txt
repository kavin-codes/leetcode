nums = [-7, -3, 2, 3, 11]

class Solution:
    def single(self, nums):
        # Initialize left pointer at the beginning of the array
        left = 0
        
        # Initialize right pointer at the end of the array
        right = len(nums) - 1
        
        # Position pointer to fill the result array from the end
        # (because we place the largest squares first)
        pos = len(nums) - 1
        
        # Create result array of same size filled with 0s
        res = [0] * len(nums)

        # Continue until left pointer crosses right pointer
        while left <= right:
            
            # Calculate square of left element
            left_sq = nums[left] * nums[left]
            
            # Calculate square of right element
            right_sq = nums[right] * nums[right]

            # Compare squares
            # The larger square should go at current 'pos'
            if left_sq > right_sq:
                res[pos] = left_sq   # Place larger square
                left += 1            # Move left pointer forward
            else:
                res[pos] = right_sq  # Place larger square
                right -= 1           # Move right pointer backward

            # Move position pointer backward
            pos -= 1

        # Return sorted squared array
        return res


s1 = Solution()
print(s1.single(nums))

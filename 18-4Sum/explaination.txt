class Solution:
    def fourSum(self, nums, target):
        # Step 1: Sort the array
        nums.sort()
        
        n = len(nums)
        result = []

        # Step 2: First loop → pick first number
        for i in range(n - 3):

            # Skip duplicate values for i
            if i > 0 and nums[i] == nums[i - 1]:
                continue

            # Step 3: Second loop → pick second number
            for j in range(i + 1, n - 2):

                # Skip duplicate values for j
                if j > i + 1 and nums[j] == nums[j - 1]:
                    continue

                # Step 4: Two pointers
                left = j + 1
                right = n - 1

                while left < right:
                    total = nums[i] + nums[j] + nums[left] + nums[right]

                    # Case 1: Found valid quadruplet
                    if total == target:
                        result.append([nums[i], nums[j], nums[left], nums[right]])

                        left += 1
                        right -= 1

                        # Skip duplicates
                        while left < right and nums[left] == nums[left - 1]:
                            left += 1

                        while left < right and nums[right] == nums[right + 1]:
                            right -= 1

                    # Case 2: Increase sum
                    elif total < target:
                        left += 1

                    # Case 3: Decrease sum
                    else:
                        right -= 1

        return result


# 🔽 Function Call / Driver Code

# Create object of Solution class
sol = Solution()

# Example input
nums = [1, 0, -1, 0, -2, 2]
target = 0

# Call the function
output = sol.fourSum(nums, target)

# Print result
print(output)

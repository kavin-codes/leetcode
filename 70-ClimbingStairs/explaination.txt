class Solution:
    def climbStairs(self, n: int) -> int:
        
        # If there are 1 or 2 steps,
        # the number of ways is equal to n itself.
        # n = 1 → 1 way
        # n = 2 → 2 ways
        if n <= 2:
            return n
        
        # first represents ways to reach step 1
        first = 1
        
        # second represents ways to reach step 2
        second = 2
        
        # Start calculating from step 3 up to step n
        for _ in range(3, n + 1):
            
            # Current step ways = previous step + step before previous
            # This follows Fibonacci pattern
            third = first + second
            
            # Move the window forward:
            # first becomes old second
            first = second
            
            # second becomes current calculated value
            second = third
        
        # second now contains the number of ways to reach step n
        return second

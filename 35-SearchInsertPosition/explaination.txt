class Solution:
    def searchInsert(self, nums, target):
        # Initialize two pointers for binary search
        left = 0
        right = len(nums) - 1
        
        # Continue searching while left pointer does not cross right
        while left <= right:
            # Find the middle index
            mid = (left + right) // 2
            
            # Case 1: Target found at mid
            if nums[mid] == target:
                return mid
            
            # Case 2: Target is greater → ignore left half
            elif nums[mid] < target:
                left = mid + 1
            
            # Case 3: Target is smaller → ignore right half
            else:
                right = mid - 1
        
        # If target not found, 'left' is the correct insert position
        return left


# 🔹 Function Call
nums = [1, 3, 5, 6]
target = 2

obj = Solution()
result = obj.searchInsert(nums, target)

print("Insert position:", result)

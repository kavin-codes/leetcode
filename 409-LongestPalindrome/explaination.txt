
s = "abccccdd"

class Solution:
    def single(self, s):

        # Dictionary to store frequency of each character
        freq = {}

        # Loop through every character in the string
        for ch in s:
            # If character already exists in dictionary
            if ch in freq:
                # Increase its count
                freq[ch] += 1
            else:
                # First occurrence of character
                freq[ch] = 1

        # Variable to store longest palindrome length
        length = 0

        # Flag to check if any odd frequency exists
        odd_found = False

        # Loop through all character counts
        for count in freq.values():
            # If count is even, use it completely
            if count % 2 == 0:
                length += count
            else:
                # If count is odd, use (count - 1)
                # so that it becomes even
                length += count - 1

                # Mark that an odd count exists
                odd_found = True

        # If at least one odd count exists,
        # we can place one character in the center
        if odd_found:
            length += 1

        # Return the maximum palindrome length
        return length

        seen.remove(ch)

Dictionary method:

Command: Count → Pair → Center

Logic: Mathematical, frequency-based

Set method:

Command: Match → Remove → Count

Logic: Greedy pairing

Both give same correct answer.
